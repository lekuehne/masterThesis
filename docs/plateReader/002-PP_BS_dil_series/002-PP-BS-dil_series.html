002-PP-BS-dil_series

Short Description

This is a the batch culture well plate experiment, with serial dilutions of starting densities of B. subtilis and P. putida monocultures. Starting densities ranged from 10-1 to 10-6 and 10-2 to 10-7 OD600, respectively. Culturing was carried out on plate reader with continuous measurements (every 1h) for ~24 h.

summary_df <- OD_long %>%

Growth Rates

fit_gc <- function(df) {
  tryCatch({
    growthcurver::SummarizeGrowth(
      data_t = as.numeric(df$time),
      data_n = as.numeric(df$mean_OD),
      bg_correct = "min"
    )
  }, error = function(e) {
    message("Failed fit")
    return(NULL)
  })
}

gc_fits <- summary_df %>%
  group_by(species, dilution) %>%
  nest() %>%
  mutate(gc = map(data, fit_gc))

Apply per group

gc_params <- summary_df %>%
  group_by(species, dilution) %>%
  nest() %>%
  mutate(
    gc = map(data, fit_gc),
    t_mid = map_dbl(gc, ~ ifelse(is.null(.x), NA, .x$vals$t_mid)),
    r     = map_dbl(gc, ~ ifelse(is.null(.x), NA, .x$vals$r)),
    k     = map_dbl(gc, ~ ifelse(is.null(.x), NA, .x$vals$k))
  ) %>%
  ungroup()


bg_df <- summary_df %>%
  group_by(species, dilution) %>%
  summarise(bg = min(mean_OD, na.rm = TRUE), .groups = "drop")

gc_params <- gc_params %>%
  left_join(bg_df, by = c("species", "dilution"))

Generate smooth fitted curves

predict_growth <- function(species_name) {
  gc_params %>%
    filter(species == species_name) %>%
    filter(!is.na(k), !is.na(r), !is.na(t_mid)) %>%
    rowwise() %>%
    mutate(
      pred = list({
        time_seq <- seq(0, max(summary_df$time), length.out = 100)
        pred_vals <- k / (1 + exp(-r * (time_seq - t_mid)))
        tibble(time = time_seq, pred_OD = pred_vals + bg)
      })
    ) %>%
    ungroup() %>%
    select(dilution, pred) %>%
    unnest(pred)
}

PP_pred <- predict_growth("Pp")
BS_pred <- predict_growth("Bs")

analysis_df <- gc_params %>%
  select(species, dilution, t_mid) %>%
  rename(t0 = t_mid)   # optional: keep old naming if you like

t0_points <- gc_params %>%
  filter(species == "Pp") %>%
  filter(!is.na(k), !is.na(r), !is.na(t_mid)) %>%
  rowwise() %>%
  mutate(
    OD_t0 = list({
      # Predicted OD at t_mid
      pred_OD <- k / (1 + exp(-r * (t_mid - t_mid))) + bg

  ungroup() %>%
  select(dilution, t_mid, OD_t0, start_OD_theoretical)
