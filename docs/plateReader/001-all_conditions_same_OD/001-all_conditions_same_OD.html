001-all_conditions_same_OD

Short Description

This is the first 96-well-plate experiment. All combinations of monoculture, and 2/3 species co-culture were tested with an initial starting OD600 of 0.001. Total well volume was 250 µl. Master mixes (MM) with OD of 0.005 were set up from overnight pre-culture and 50 µl MM used per well. 12 replicates (columns on plate) per condition. Order of conditions as follows:

Rows of 96-well-plate
A B.s.
B B.s. + NI
C NI
D P.p + B.s. + NI
E P.p + B.s.
F P.p + NI
G P.p
H control (M9)

Soil Preparation for NI was done in H2O only. No Preculturing of natural inocoulum, no centrifugation. 1g soil (form TER lab course ’26) + 10ml H2O were vortexted for 10 min and subsequently vacuum filtered through filter paper. Yield was very low. Cells were pelleted and resuspended in 2 ml M9 media.

process_plate <- function(file_path) {
  
  raw <- read_excel(file_path, col_names = FALSE)
  
  # Extract plate sections
  OD600  <- raw[25:32, 2:13]
  GFP100 <- raw[57:64, 2:13]
  GFP80  <- raw[89:96, 2:13]
  RFP100 <- raw[121:128, 2:13]
  RFP80  <- raw[153:160, 2:13]
  
  # Convert to long format
  to_long <- function(df, name) {
    out <- as.data.frame(as.table(as.matrix(df)))
    out$wavelength <- name
    return(out)
  }
  
  OD600  <- to_long(OD600,  "OD600")
  GFP100 <- to_long(GFP100, "GFP100")
  GFP80  <- to_long(GFP80,  "GFP80")
  RFP100 <- to_long(RFP100, "RFP100")
  RFP80  <- to_long(RFP80,  "RFP80")
  
  # Combine
  plate_concat <- rbindlist(list(OD600, GFP100, GFP80, RFP100, RFP80))
  setnames(plate_concat, c("row", "col", "value", "wavelength"))
  
  plate_concat$value <- as.numeric(as.character(plate_concat$value))
  plate_concat$well <- paste0(plate_concat$row, plate_concat$col)
  
  # Date + time from same file
  date <- read_excel(file_path, range = "B5", col_names = FALSE)[[1]]
  time <- read_excel(file_path, range = "B6", col_names = FALSE)[[1]]
  
  plate_concat$datetime <- strptime(
    paste(date, time),
    format = "%Y-%m-%d %I:%M:%S %p"
  )
  
  # Conditions
  plate_concat$condition <- dplyr::recode(
    plate_concat$row,
    "A" = "Bs", "B" = "Bs+Soil", "C" = "Soil",
    "D" = "Bs+Pp", "E" = "Bs+Pp+Soil",
    "F" = "Pp", "G" = "Pp+Soil", "H" = "M9"
  )
  
  # Final tidy
  plate_concat <- plate_concat[, .(well, datetime, condition, wavelength, value)]
  
  return(plate_concat)
}
